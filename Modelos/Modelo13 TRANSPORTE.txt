  function limpiarCategoriasBlog() {
    $('.premium-blog-post-outer-container').each(function() {
      var $container = $(this).find('.premium-blog-post-categories');

      if ($container.length) {
        let $links = $container.find('a');
        let filteredLinks = [];

        $links.each(function() {
          let text = $(this).text().trim().toLowerCase();
          if (text !== 'uncategorized' && filteredLinks.length < 3) {
            filteredLinks.push($(this));
          }
        });

        // Limpiar contenido anterior
        $container.html('<i class="fa fa-align-left fa-fw" aria-hidden="true"></i>');

        // Agregar solo los primeros 3 (sin uncategorized)
        filteredLinks.forEach(function($link, index) {
          if (index > 0) $container.append(' ');
          $container.append($link);
        });
      }
    });
  }

  // Ejecutar al inicio
  limpiarCategoriasBlog();

  // Ejecutar al hacer clic en paginación
  $(document).on('click', '.page-numbers', function() {
    // Usamos un timeout para esperar que se cargue el nuevo contenido (en caso de AJAX)
    setTimeout(() => {
      limpiarCategoriasBlog();
    }, 500); // ajusta si el contenido carga más lento
  });

});</script>

  const parent = document.querySelector('.disp-tabs');
  const scrollAmount = 200;

  if (!container || !parent) return;
	
  // Detect click en flechas usando ::before y ::after como pseudo-elementos
  parent.addEventListener('click', function (e) {
    const bounds = parent.getBoundingClientRect();
    const x = e.clientX - bounds.left;

    if (x < 40) {
      // click en flecha izquierda
      container.scrollBy({ left: -scrollAmount, behavior: 'smooth' });
    } else if (x > bounds.width - 40) {
      // click en flecha derecha
      container.scrollBy({ left: scrollAmount, behavior: 'smooth' });
    }
  });
	
	  function updateArrows() {
    const maxScroll = container.scrollWidth - container.clientWidth;

    if (container.scrollLeft <= 0) {
      parent.classList.add('no-left');
    } else {
      parent.classList.remove('no-left');
    }

    if (container.scrollLeft >= maxScroll - 1) {
      parent.classList.add('no-right');
    } else {
      parent.classList.remove('no-right');
    }
  }
	
	  // Ejecutar al inicio y cada vez que se hace scroll o resize
  container.addEventListener('scroll', updateArrows);
  window.addEventListener('resize', updateArrows);
  updateArrows();
});
